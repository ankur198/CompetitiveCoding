1) Input list = ['kasol', 'kasol', 'manali', 'delhi', 'delhi', 'manali, 'kasol']
Output = 
kasol - 3
manali - 2
delhi - 2

2) Input list = ['1', 'a', 'c', 'b', 'c' , '1']

Output =  1 - 2
a - 1
c - 2
b - 1
